static Future<bool> isCommandAvailable(String command) async {
  try {
    final result = await runCommand(
      Platform.isWindows ? 'where' : 'which',
      [command],
      showOutput: false,
    );
    return result.exitCode == 0;
  } catch (e) {
    return false;
  }
}
